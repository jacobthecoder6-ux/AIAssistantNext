                return "Hello! How can I assist you today?";
            } else if (userMessage.includes('how are you')) {
                return "I'm doing well, thank you! How are you feeling?";
            } else if (userMessage.includes('what is your name')) {
                return "I'm a multilingual AI chatbot, here to help!";
            }

            // Code-Related Requests
            else if (userMessage.includes('code') || userMessage.includes('program') || userMessage.includes('script')) {
                return generateCode(userMessage);
            }

            // Translation Requests
            else if (userMessage.includes('translate')) {
                return "I can help with translations! Please provide the text you'd like translated and the languages.";
            }

            // Explanation Requests
            else if (userMessage.includes('explain') || userMessage.includes('what is') || userMessage.includes('how does')) {
                return "I can definitely help with that! Could you please provide more details about what you'd like me to explain?";
            }

            // Creative Writing Requests
            else if (userMessage.includes('write a story') || userMessage.includes('write a poem')) {
                return "I can try my hand at some creative writing! What kind of story or poem would you like?";
            }

            // General Knowledge
            else if (userMessage.includes('what is the capital of') || userMessage.includes('what is the population of')) {
                return "I can provide you with that information. Please specify the country or region you're interested in.";
            }

            // Apologize for sensitive topics
            else if (userMessage.includes('sex') || userMessage.includes('violence') || userMessage.includes('hate')) {
                return "I'm sorry, but I can't provide information on that topic. I am not designed to discuss sensitive content.";
            }

            // Polite and Encouraging Default
            else {
                return "That's an interesting topic! I'm still under development, but I'll do my best to help. Could you tell me more about what you're looking for?";
            }
        }
